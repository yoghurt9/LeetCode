    public static boolean hasAlternatingBits(int n){
//        if (n==1||ntrue;
        int pre = n&1;
        n>>>=1;
        while (n!=0){
            if ((n&1)==pre)
                return false;
            pre = n&1;
            n>>>=1;
        }
        return true;
    }
